# sentence = '안녕하세요 홍길동님 만나서 반갑습니다'
  - `print(sentence)` => 안녕하세요 홍길동님 만나서 반갑습니다
# f{}을 이용해 input 값 바꾸기
  - `name = '홍길동'`
  - `sentence = f'안녕하세요 {name}님 만나서 반갑습니다'`
  - `print(sentence)` => 안녕하세요 홍길동님 만나서 반갑습니다
 # f{}--formatting은 여러 변수에도 사용 가능함
  - `month = '12월'`
  - `day = '25일'`
  - `sentence = f'오늘은 {month} {day} 입니다'`
  - `print(sentence)` => 오늘은 12월 25일 입니다
